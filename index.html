Команды YT-DLP

# Скачать только аудио в MP3  
yt-dlp -f "bestaudio" --extract-audio --audio-format mp3 -o "~/Desktop/%(title)s.%(ext)s" "

# Скачать лучшее качество видео + аудио  
yt-dlp -f "bestvideo+bestaudio" -o "~/Desktop/%(title)s.%(ext)s" "

# Скачать лучшее качество видео + аудио и сохранить в MKV  
yt-dlp -f "bestvideo+bestaudio" --merge-output-format mkv -o "~/Desktop/%(title)s.%(ext)s" "

# Скачать только лучшее качество видео  
yt-dlp -f "bestvideo" -o "~/Desktop/%(title)s.%(ext)s" "
